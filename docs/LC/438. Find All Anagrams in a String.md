Approach 1:

Pseudocode:
- Define array result, counter occurrences of each character in p, s
- left pointer = 0
- Iterate in s for each character as right pointer
	- s_count[s[right]] += 1
	- Validate window length == length of p
		- if right - left + 1 > p:
				if s_count[s[left]] == 1:
                    del s_count[s[left]]
                else:
                    s_count[s[left]] -= 1
                left += 1
            compare two counter: if s_count == p_count -> result.append(left)

- return result 

```python
from collections import Counter
from typing import List

class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        result = []
        p_count = Counter(p)
        s_count = Counter()
        
        left = 0
        for right in range(len(s)):
            s_count[s[right]] += 1
            
            # Ensure the window length equals the length of p
            if right - left + 1 > len(p):
                if s_count[s[left]] == 1:
                    del s_count[s[left]]
                else:
                    s_count[s[left]] -= 1
                left += 1
            
            # Compare the two counters
            if s_count == p_count:
                result.append(left)
        
        return result
```
